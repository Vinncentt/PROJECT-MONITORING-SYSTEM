// TABLES CREATING


CREATE TABLE IF NOT EXISTS Task(
	id VARCHAR(50),
    Label VARCHAR(50),
    Description TEXT,
    StartDate DATE,
    EndDate DATE,
	Status VARCHAR(50),
    Price FLOAT,
	Facturated BOOL,
	Payed BOOL,
    PRIMARY KEY (id)
);



CREATE TABLE IF NOT EXISTS Project (
	id INT,
    Name VARCHAR(50),
    Description TEXT,
    StartDate DATE,
	Status VARCHAR(50),
    Price FLOAT,
    PRIMARY KEY (id)
);

CREATE TABLE IF NOT EXISTS Document(
	id INT,
    Path TEXT,
    PRIMARY KEY (id)
);


CREATE TABLE IF NOT EXISTS Employee(
	id VARCHAR(50),
    Fname VARCHAR(50),
    Lname VARCHAR(50),
	Phone VARCHAR(50),
	Email VARCHAR(50),
	Password VARCHAR(50),
    PRIMARY KEY (id)
);

CREATE TABLE IF NOT EXISTS Mediator(
	id VARCHAR(50),
    Name VARCHAR(50),
    ContactName TEXT,
    Phone VARCHAR(50),
    Adress VARCHAR(50),
	ContactEmail VARCHAR(100),
    Website TEXT,
    PRIMARY KEY (id)
);

CREATE TABLE IF NOT EXISTS Delivery(
	id VARCHAR(50),
    Label VARCHAR(50),
    Description TEXT,
    PRIMARY KEY (id)
);

CREATE TABLE IF NOT EXISTS Profile(
	id INT,
    Labels VARCHAR(50),
    PRIMARY KEY (id)
);

CREATE TABLE IF NOT EXISTS Work(
	id INT,
    StartDate DATE,
    EndDate DATE,
    PRIMARY KEY (id)
);
